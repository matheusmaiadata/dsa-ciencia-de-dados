# Projeto 9 - Chatbot Personalizado com Sistema de Recomendação Para Suporte ao Cliente Usando LLMs

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsabusp9 python=3.13

# Ative o ambiente:

conda activate dsabusp9 (ou: source activate dsabusp9)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Execute os scripts:

streamlit run projeto9.py
